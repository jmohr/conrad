from nose.tools import raises

from conrad import models, fields, db
# 
class TestModel:
# 
    def test_init(self):
# 
        class MyModel(models.Model):
            pass
        assert MyModel in models.cache
# 
    def test_fields(self):
# 
        class MyModel(models.Model):
            name = fields.Char()
            age = fields.Integer()
            birthdate = fields.Date()
            address = fields.Text()
            active = fields.Boolean()
        assert isinstance(MyModel._fields, dict), '_fields is not a dict'
        for field in ['name', 'age', 'birthdate', 'address', 'active']:
            assert field in MyModel._fields, '%s not in _fields' % field
            assert isinstance(MyModel._fields[field], fields.Field), '%s is not Field' % field
# 
    def test_meta(self):
# 
        class MyModel(models.Model):
            pass
        assert hasattr(MyModel, 'Meta')
        for attr in ['pk', 'db', 'table']:
            assert hasattr(MyModel.Meta, attr), 'Meta does not have default %s' % attr
            assert getattr(MyModel.Meta, attr) is not None, 'Meta.%s is None' % attr
        assert MyModel.Meta.pk == 'id'
        assert MyModel.Meta.table == '`mymodel`', 'Meta.table %s is not `mymodel`' % (
                MyModel.Meta.table)
        assert MyModel.Meta.db == db.database

    @raises(ValueError)
# 
    def test_meta_db(self):
# 
        class MyModel(models.Model):
# 
            class Meta:
                db = 'bad'
# 
    def test_escape(self):
        tests = {
            'StudlyCapsGoHere':'`studlycapsgohere`',
            'camelCaseGoesHere':'`camelcasegoeshere`',
            'some_underscores_perhaps':'`some_underscores_perhaps`',
            'and, punctuation? YEAH!':'`and_punctuation_yeah`'
        }
        for orig, expected in tests.items():
            escaped = models.ModelBase.escape(orig)
            assert escaped == expected, 'Escape: %s is not %s' % (
                escaped, expected)
