import pyodbc

from conrad.adapters.odbc import ODBC
# 
class TestODBC:
# 
    def setup(self):
        self.dsn_str = 'DRIVER={SQLite3};DATABASE=/tmp/test.db'
# 
    def test_init_dsn(self):
        cxn = ODBC()
        cxn.connect(self.dsn_str)
        assert cxn is not None
        assert isinstance(cxn.connection, pyodbc.Connection)
# 
    def test_execute(self):
        cxn = ODBC()
        cxn.connect(self.dsn_str)
        cxn.execute('DROP TABLE IF EXISTS test_table_1')
        cxn.execute('CREATE TABLE `test_table_1` (id integer, name varchar(255))')
        res = cxn.execute('SELECT * FROM test_table_1')
        assert isinstance(res, list)
        cxn.execute('INSERT INTO test_table_1 (id, name) VALUES (?, ?)', 123, 'foobar')
        res = cxn.execute('SELECT * FROM test_table_1')
        assert isinstance(res, list)
        assert len(res) == 1
        assert 'foobar' in res[0]
