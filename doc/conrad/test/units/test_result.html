from conrad.result import Result
# 
class TestResult(object):
# 
    def test_init(self):
        r = Result(name='Old Guy', age=90)
        assert r is not None
# 
    def test_getitem(self):
        r = Result(name='Old Guy', age=90)
        assert r['name'] == 'Old Guy'
        assert r['age'] == 90
# 
    def test_setitem(self):
        r = Result(name='Old Guy', age=90)
        r['name'] = 'Young Guy'
        r['age'] = 18
        assert r['name'] == 'Young Guy'
        assert r['age'] == 18
